条件语句：

```sql
delimiter //
create procedure proc_if ()
begin
	declare I int default 0;
	if i = 1 then
		select 1;
	elseif i = 2 then
		select 2;
	else
		select 7;
	end if;
end//
delimiter ;
```

循环语句：

while循环

```sql
delimiter //
create procedure proc_while()
begin
	declare num int;
	set num = 0;
	while num < 10 do
		select num;
		set num = num + 1;
	end while;
end //
delimiter ;
```

repeat循环

```sql
delimiter //
create procedure proc_repeat()
begin
	declare i int;
	set i = 0;
	repeat
		select i;
		set i = i + 1;
		until i >= 5
	end repeat;
end //
delimiter ;
```

loop循环

```sql
delimiter //
create procedure proc_loop()
begin
	declare i int default 0;
	loop_label: loop
		set i = i + 1;
		if i < 8 then
			iterate loop_lable;
		end if;
		if i >= 10 then
			leave loop_label;
		end if;
		select i;
	end loop loop_label;
end //
delimiter ;
```
